# Function to detect the bottom of the screen
function __notbottom() {
  local pos
  IFS='[;' read -p $'\e[6n' -d R -a pos -rs || echo "failed with error: $? ; ${pos[*]}"
  CURLN=$((${pos[1]} +1 ))
  if [ "$CURLN" -ge "$LINES" ] ; then return -1 ; fi
}
